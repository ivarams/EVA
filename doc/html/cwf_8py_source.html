158  def parse_file_recognition_output(self, lines):
159  """!
160  @brief Parse standard output containing time dimensions from NetCDF
161  files into a structured format.
162  @returns A list of dictionaries with file and time dimension information.
163  """
164  result = []
165  for line in lines:
166  # Each output line looks like this:
167  # /tmp/meteo20160606_00.nc time = "2016-06-06 12", "2016-06-06 15", "2016-06-06 18", "2016-06-06 21", "2016-06-07" ;
168  if len(line) < 5:
169  continue
170  data = {}
171  tokens = line.split()
172  data['path'] = tokens[0]
173  time_str = ' '.join(tokens[3:-1])
174  times = sorted([eva.netcdf_time_to_timestamp(x.strip(' "')) for x in time_str.split(',')])
175  data['time_steps'] = times
176  result += [data]
177  return result
