484

488  * Send the Job for execution to the Executor

492  # Checks for real Productstatus events from the message queue

495  # Only process messages with the correct version
496  if event.protocol_version()[0] != 1:
497  self.logger.warning('Event version is %s, but I am only accepting major version 1. Discarding message.', '.'.join(event.protocol_version()))

505  self.statsd.incr('resource_object_version_too_old')
506  self.remove_event_from_queues(event)
507  return

509  # Create event job if it has not been created yet
510  if not hasattr(event, 'job'):
511  self.logger.debug('Start processing event: %s', event)
512  self.instantiate_productstatus_data(event)
513  resource = event.data
514  if not self.adapter.validate_resource(resource):
515  self.logger.debug('Adapter did not validate the current event, skipping.')
516  self.statsd.incr('productstatus_rejected_events')
517  self.remove_event_from_queues(event)
518  else:
519  self.logger.debug('Creating a Job object for the current event...')
520  self.statsd.incr('productstatus_accepted_events')
521  event.job = self.adapter.create_job(event.id(), resource)

523  self.logger.debug('No Job object was returned by the adapter; assuming no-op.')
524  self.remove_event_from_queues(event)
525  else:

527  event.job.timer = self.statsd.timer('execution_time')
528  event.job.logger.info('Created Job object for event: %s', event)
529 
530  # Start job if it is not running

532  event.job.logger.info('Sending job to executor for asynchronous execution...')
533  event.job.timer.start()

538  elif event.job.started():

540  event.job.logger.debug('Job is running, polling executor for job status...')
541  self.executor.sync(event.job)
542  event.job.logger.debug('Finished polling executor for job status.')

544  # Tell adapter that the job has finished

550  event.job.logger.info('Adapter has finished processing the job.')
