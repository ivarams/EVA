193  def parse_file_recognition_output(self, lines):
194  """!
195  @brief Parse standard output containing time dimensions from NetCDF
196  files into a structured format.
197  @returns A list of dictionaries with file and time dimension information.
198  """
199  result = []
200  for line in lines:
201  # Each output line looks like this:
202  # /tmp/meteo20160606_00.nc time = "2016-06-06 12", "2016-06-06 15", "2016-06-06 18", "2016-06-06 21", "2016-06-07" ;
203  if len(line) < 5:
204  continue
205  data = {}
206  tokens = line.split()
207  data['path'] = tokens[0]
208  data['extension'] = os.path.splitext(data['path'])[1]
209  if self.is_netcdf_data_output(data):
210  time_str = ' '.join(tokens[3:-1])
211  times = sorted([eva.netcdf_time_to_timestamp(x.strip(' "')) for x in time_str.split(',')])
212  data['time_steps'] = times
213  result += [data]
214  return result

257  if lifetime_index < len(self.env['EVA_CWF_LIFETIME']):

265  data_instance.expires = self.expiry_from_hours(min(self.env['EVA_CWF_LIFETIME']))

269  resources['datainstance'] += [data_instance]
